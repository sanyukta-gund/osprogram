/*
Implement the C program for Disk Scheduling Algorithms: SSTF, SCAN, C-Look
considering the initial head position moving away from the spindle.
*/

/*
nano disk_scheduling.c
gcc disk_scheduling.c -o disk_scheduling
./disk_scheduling
*/


#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void sstf(int request[], int n, int head) {
    int visited[n];
    for(int i = 0; i < n; i++) visited[i] = 0;

    int total = 0, curr = head;
    for(int i = 0; i < n; i++) {
        int min_dist = 100000, index = -1;
        for(int j = 0; j < n; j++) {
            if(!visited[j] && abs(request[j] - curr) < min_dist) {
                min_dist = abs(request[j] - curr);
                index = j;
            }
        }
        visited[index] = 1;
        total += abs(request[index] - curr);
        curr = request[index];
    }
    printf("SSTF Total Head Movement: %d\n", total);
}

void scan(int request[], int n, int head, int disk_size) {
    int total = 0, curr = head;
    int left[n], right[n], lcount = 0, rcount = 0;

    for(int i = 0; i < n; i++) {
        if(request[i] < head) left[lcount++] = request[i];
        else right[rcount++] = request[i];
    }

    for(int i = 0; i < lcount-1; i++)
        for(int j = 0; j < lcount-i-1; j++)
            if(left[j] > left[j+1]) { int temp = left[j]; left[j] = left[j+1]; left[j+1] = temp; }
    for(int i = 0; i < rcount-1; i++)
        for(int j = 0; j < rcount-i-1; j++)
            if(right[j] > right[j+1]) { int temp = right[j]; right[j] = right[j+1]; right[j+1] = temp; }

    for(int i = 0; i < rcount; i++) { total += abs(curr - right[i]); curr = right[i]; }
    if(lcount > 0) { total += abs(curr - (disk_size-1)); curr = disk_size-1; }
    for(int i = lcount-1; i >=0; i--) { total += abs(curr - left[i]); curr = left[i]; }

    printf("SCAN Total Head Movement: %d\n", total);
}

void clook(int request[], int n, int head) {
    int total = 0, curr = head;
    int left[n], right[n], lcount = 0, rcount = 0;

    for(int i = 0; i < n; i++) {
        if(request[i] < head) left[lcount++] = request[i];
        else right[rcount++] = request[i];
    }

    for(int i = 0; i < lcount-1; i++)
        for(int j = 0; j < lcount-i-1; j++)
            if(left[j] > left[j+1]) { int temp = left[j]; left[j] = left[j+1]; left[j+1] = temp; }
    for(int i = 0; i < rcount-1; i++)
        for(int j = 0; j < rcount-i-1; j++)
            if(right[j] > right[j+1]) { int temp = right[j]; right[j] = right[j+1]; right[j+1] = temp; }

    for(int i = 0; i < rcount; i++) { total += abs(curr - right[i]); curr = right[i]; }
    if(lcount > 0) { total += abs(curr - left[0]); curr = left[0]; }
    for(int i = 1; i < lcount; i++) { total += abs(curr - left[i]); curr = left[i]; }

    printf("C-LOOK Total Head Movement: %d\n", total);
}

int main() {
    int n, head, disk_size;
    printf("Enter number of requests: ");
    scanf("%d", &n);
    int request[n];
    printf("Enter request sequence: ");
    for(int i = 0; i < n; i++) scanf("%d", &request[i]);
    printf("Enter initial head position: ");
    scanf("%d", &head);
    printf("Enter disk size: ");
    scanf("%d", &disk_size);

    sstf(request, n, head);
    scan(request, n, head, disk_size);
    clook(request, n, head);

    return 0;
}
