#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int pid;
    int arrival_time;
    int burst_time;
    int remaining_time;
    int waiting_time;
    int turnaround_time;
} Process;


int find_srtf(Process p[], int n, int t) {
    int min_index = -1;
    int min_time = 1e9;
    for(int i = 0; i < n; i++) {
        if(p[i].arrival_time <= t && p[i].remaining_time > 0) {
            if(p[i].remaining_time < min_time) {
                min_time = p[i].remaining_time;
                min_index = i;
            }
        }
    }
    return min_index;
}


void srtf(Process p[], int n) {
    int completed = 0, t = 0;
    int total_wt = 0, total_tat = 0;

    for(int i = 0; i < n; i++)
        p[i].remaining_time = p[i].burst_time;

    while(completed < n) {
        int idx = find_srtf(p, n, t);
        if(idx != -1) {
            p[idx].remaining_time--;
            if(p[idx].remaining_time == 0) {
                completed++;
                p[idx].turnaround_time = t + 1 - p[idx].arrival_time;
                p[idx].waiting_time = p[idx].turnaround_time - p[idx].burst_time;
                total_wt += p[idx].waiting_time;
                total_tat += p[idx].turnaround_time;
            }
        }
        t++;
    }

    printf("\nSRTF Scheduling:\n");
    printf("PID\tAT\tBT\tWT\tTAT\n");
    for(int i = 0; i < n; i++)
        printf("%d\t%d\t%d\t%d\t%d\n", p[i].pid, p[i].arrival_time, p[i].burst_time, p[i].waiting_time, p[i].turnaround_time);

    printf("Average Waiting Time: %.2f\n", (float)total_wt / n);
    printf("Average Turnaround Time: %.2f\n", (float)total_tat / n);
}


void round_robin(Process p[], int n, int quantum) {
    int t = 0, completed = 0, total_wt = 0, total_tat = 0;
    int queue[n]; 
    int front = 0, rear = 0;
    int remaining[n];

    for(int i = 0; i < n; i++) {
        remaining[i] = p[i].burst_time;
    }

    int visited[n];
    for(int i = 0; i < n; i++) visited[i] = 0;

    printf("\nRound Robin Scheduling (Quantum = %d):\n", quantum);
    printf("PID\tAT\tBT\tWT\tTAT\n");

    t = 0;
    while(completed < n) {
        int done = 1;
        for(int i = 0; i < n; i++) {
            if(p[i].arrival_time <= t && remaining[i] > 0) {
                done = 0;
                if(remaining[i] > quantum) {
                    remaining[i] -= quantum;
                    t += quantum;
                } else {
                    t += remaining[i];
                    remaining[i] = 0;
                    p[i].turnaround_time = t - p[i].arrival_time;
                    p[i].waiting_time = p[i].turnaround_time - p[i].burst_time;
                    total_wt += p[i].waiting_time;
                    total_tat += p[i].turnaround_time;
                    completed++;
                }
            }
        }
        if(done) t++;
    }

    for(int i = 0; i < n; i++)
        printf("%d\t%d\t%d\t%d\t%d\n", p[i].pid, p[i].arrival_time, p[i].burst_time, p[i].waiting_time, p[i].turnaround_time);

    printf("Average Waiting Time: %.2f\n", (float)total_wt / n);
    printf("Average Turnaround Time: %.2f\n", (float)total_tat / n);
}

int main() {
    int n, quantum;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    Process p[n];
    for(int i = 0; i < n; i++) {
        p[i].pid = i+1;
        printf("Enter arrival time and burst time for P%d: ", i+1);
        scanf("%d %d", &p[i].arrival_time, &p[i].burst_time);
        p[i].remaining_time = p[i].burst_time;
    }

    
    srtf(p, n);

    
    for(int i = 0; i < n; i++) {
        p[i].waiting_time = 0;
        p[i].turnaround_time = 0;
        p[i].remaining_time = p[i].burst_time;
    }

    printf("\nEnter time quantum for Round Robin: ");
    scanf("%d", &quantum);

    
    round_robin(p, n, quantum);

    return 0;
}
